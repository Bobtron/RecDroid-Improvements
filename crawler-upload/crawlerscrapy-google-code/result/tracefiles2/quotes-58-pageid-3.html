ruirui

Android MVC Framework

Introduction

brian-android는 안드로이드 개발 시 반복적인 코드 사용을 줄여서 좀 더 빠르고 편리한 안드로이드 개발 지원하고자 만든 framework입니다. MVC, Dependency Injection, Event Handling 등의 기능을 제공하여 개발자가 UI 관련 요소에 신경을 덜 쓰도록하고, 로직 개발에 좀더 집중할 수 있도록 도와 줍니다. 빠른 프로토타이핑이 필요할 경우나 UI와 로직을 병렬로 개발할 경우 유용하게 쓰였으면 좋겠습니다.

Features

    @Activity(layout=R.layout.view_note) public void viewNote() { } }

/** * as-was android code * / public void Class1 extends Activity { @Override protected void onCreate(Bundle savedInstanceState) { super.onCreate(savedInstanceState); setContentView(R.layout.create_note); } } public void Class2 extends Activity { @Override protected void onCreate(Bundle savedInstanceState) { super.onCreate(savedInstanceState); setContentView(R.layout.view_note); } } * @Inject 를 이용한 Dependency Injection public class SimpleNotepadMain extends AbstractMethodActivity { /* notepadBo is injected from beanFactory */ @Inject("notepadBo") private NotepadBo notepadBo;

@Activity(layout=R.layout.main)
public void simpleNotepadMain() {
    List<Note> notes = notepadBo.getNotes();
    // do something
}

}

/** * declare beans * */ public class MyBeanConfig extends AbstractBeanConfig { @Override protected void initializeBeans() { BeanProxy dao = addBean("notepadDao", MemoryNotepadDao.class); BeanProxy noteBo = addBean("notepadBo", NotepadBo.class, dao); } } * @Event 를 이용한 Event Handling @Event(view=R.id.btnCreateNote, event=OnClickListener.class) public void doBtnCreateNote() throws IllegalActivityMethodNameException { // do something here }

/*
 * @Event attribute "view" and "event" can be omitted using code convention. ie) Event Name + View Id
 * onClickBtnCreateNote() is same as doBtnCreateNote()
 * */
@Event
public void onClickBtnCreateNote() throws IllegalActivityMethodNameException {
    // do something here
}

* redirectActivity() 를 이용한 Activity 간 이동 간소화 @Activity(layout=R.layout.create_note) public void createNote() { // do something }

@Activity(layout=R.layout.view_note)
public void viewNote() {
    int noteId = getExtraValue("noteId");
    // do something
}

/*
 * move to createNote method activity
 * */
@Event
public void onClickBtnCreateNote() throws IllegalActivityMethodNameException {
    redirectActivity("createNote");
}

/*
 * move to viewNote method activity using intent parameter noteId
 * */
@Event
public void onClickBtnViewNote() throws IllegalActivityMethodNameException {
    redirectActivity("viewNote", "noteId", noteId);
}
