public boolean checkAccount() {
    Cursor c = getContentResolver().query(SipProfile.ACCOUNT_URI,
            new String[] {
                    SipProfile.FIELD_ID,
                    SipProfile.FIELD_ACC_ID,
                    SipProfile.FIELD_REG_URI,
                    SipProfile.FIELD_DATA

            }, null, null, SipProfile.FIELD_PRIORITY + &quot; ASC&quot;);

    if (c != null) {

        try {
            if (c.moveToFirst()) {
                SipProfile profile = new SipProfile(c);
                existingProfileId = profile.id;
                Log.v(tag, profile.display_name+&quot; &quot;+profile.acc_id+&quot; &quot;+profile.data+&quot; &quot;+profile.id);
            }
        } catch (Exception e) {
            e.printStackTrace();
            Log.v(tag, &quot;Smething went wrong&quot;);
        } finally {
            c.close();
        }

    }
    if (existingProfileId == SipProfile.INVALID_ID) {
        return true;
    }
    return false;
}
